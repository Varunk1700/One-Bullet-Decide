 2#include <string>
 3using namespace std;

 5// ABSTRACTION — Enemy is abstract. It defines what every enemy
 6// must do (isThreat, getType) but leaves the HOW to subclasses.
 7// You can never do: new Enemy() — it won't compile.
 8class Enemy {
 9protected:
10    string name;   // subclasses can read this directly
11    string hint;   // shown to the player as a clue
12public:
13    Enemy(string n, string h) : name(n), hint(h) {}
14
15    string getName() { return name; }
16    string getHint() { return hint; }
17
18    // pure virtual = every subclass MUST override these.
19    // This is what makes Enemy an abstract class.
20    virtual bool   isThreat() = 0;
21    virtual string getType()  = 0;
22    virtual ~Enemy() {}   // always virtual when using pointers
23};
24
25// INHERITANCE — DangerousEnemy gets everything Enemy has,
26// then fills in isThreat() and getType() its own way.
27class DangerousEnemy : public Enemy {
28public:
29    DangerousEnemy(string n, string h)
30        : Enemy(n, h) {}   // pass name + hint up to base
31
32    bool   isThreat() override { return true;        }
33    string getType()  override { return "DANGEROUS"; }
34};
35
36// INHERITANCE — HarmlessEnemy does the same, just returns false.
37// Same interface, different behaviour — that's polymorphism too.
38class HarmlessEnemy : public Enemy {
39public:
40    HarmlessEnemy(string n, string h)
41        : Enemy(n, h) {}
42
43    bool   isThreat() override { return false;      }
44    string getType()  override { return "HARMLESS"; }
45};

 1// ENCAPSULATION — Player keeps its own state locked away.
 2// Nothing outside this class can touch score or lives directly.
 3class Player {
 4private:
 5    int  score;       // nobody touches this from outside
 6    int  lives;       // same here — controlled via methods only
 7    bool hasBullet;   // resets each round
 8public:
 9    // start fresh every game
10    Player() : score(0), lives(3), hasBullet(true) {}
11
12    // read-only getters — we give access but not control
13    int  getScore()  { return score;     }
14    int  getLives()  { return lives;     }
15    bool isAlive()   { return lives > 0; }
16
17    // POLYMORPHISM in action — Enemy* could be Dangerous or Harmless.
18    // isThreat() resolves to the right subclass at runtime.
19    void shoot(Enemy* e) {
20        hasBullet = false;
21        if (e->isThreat()) {   // virtual dispatch happens here
22            score += 10;
23            cout << "Great shot! +10\n";
24        } else {
25            lives--;
26            cout << "Wrong call! -1 life\n";
27        }
28    }
29
30    void skip(Enemy* e) {
31        if (!e->isThreat()) {  // skipping a harmless one is smart
32            score += 5;
33            cout << "Good call! +5\n";
34        } else {
35            lives--;
36            cout << "You got hit! -1 life\n";
37        }
38    }
39
40    void resetBullet() { hasBullet = true; }  // one bullet per round
41
42    // rank based on final score — S is perfect, C means rough day
43    string getRank() {
44        if (score >= 40) return "S RANK";
45        if (score >= 25) return "A RANK";
46        if (score >= 15) return "B RANK";
47        return             "C RANK";
48    }
49};

 2    // create a fresh player — score 0, lives 3, bullet ready
 3    Player player;
 4
 5    // ABSTRACTION + POLYMORPHISM together:
 6    // We store base-class pointers. The array doesn't care whether
 7    // each slot holds Dangerous or Harmless — Enemy* covers both.
 8    Enemy* enemies[4] = {
 9        new DangerousEnemy("Goomba", "Shoot it!"),
10        new HarmlessEnemy ("Yoshi",  "Skip me!"),
11        new DangerousEnemy("Koopa",  "Danger!"),
12        new HarmlessEnemy ("Toad",   "Ally!")
13    };
14
15    for (int i = 0; i < 4; i++) {
16        if (!player.isAlive()) break;  // game over check
17
18        // these all come from the abstract Enemy interface
19        cout << enemies[i]->getName() << "\n";
20        cout << enemies[i]->getType() << "\n";
21        cout << enemies[i]->getHint() << "\n";
22
23        char ch;
24        cin >> ch;   // 's' to shoot, anything else to skip
25        if (ch == 's')
26            player.shoot(enemies[i]);  // runtime dispatch to right isThreat()
27        else
28            player.skip(enemies[i]);   // same here
29
30        player.resetBullet();   // bullet reloads each round
31    }
32
33    // getters keep score and lives private but readable — encapsulation
34    cout << player.getRank() << " — "
35         << player.getScore() << " pts\n";
36
37    // clean up — virtual destructor on Enemy handles this safely
38    for (int i = 0; i < 4; i++) delete enemies[i];
39    return 0;

OOP PRINCIPLES

📦
Classes & Objects
Encapsulation of data + behaviour
struct Enemy { string name; bool isThreat; string sprite; string describe(); };
Game entities (Enemy, Player, ResultSystem) are modelled as independent classes with private state and public interfaces — a core OOP pattern.
🔒
Encapsulation
Private data, public methods
class Player { private: bool hasBullet; int lives; public: void shoot(); void skip(); };
Player state is hidden behind a public interface — the game logic never touches hasBullet directly, enforcing encapsulation.
🧬
Inheritance
DangerousEnemy / HarmlessEnemy
class DangerousEnemy : public Enemy { ... }; class HarmlessEnemy : public Enemy { ... };
Enemy subtypes inherit base behaviour and override isThreat() — illustrating single-level inheritance with meaningful specialisation.
🎭
Polymorphism
Virtual method dispatch
virtual bool isThreat() = 0; // resolved at runtime: Enemy* e = new DangerousEnemy(); e->isThreat(); // → true
A pure virtual isThreat() on the base class lets the Game loop treat all enemies uniformly while dispatching the correct behaviour at runtime.
🎭
Abstraction
Hide complexity, expose what matters
// Enemy is abstract — can't be created directly class Enemy { virtual bool isThreat() = 0; // pure virtual virtual string getType() = 0; }; // Enemy* e = new Enemy(); ← won't compile!
Enemy is an abstract base class — it defines what every enemy must do (isThreat, getType) without specifying how. You can never create a raw Enemy; only its concrete subtypes can be instantiated.

HOW IT WORKS

01
Game Starts
Player object is initialised with 3 lives and a bullet. 5 enemies are selected at random from the enemy pool and shuffled.
gStart()
02
Enemy Appears
The Game calls nextEnemy() — an Enemy object is fetched and its sprite + description are rendered to the stage.
Enemy::describe()
03
Player Decides
Player calls either shoot() or skip(). The decision is passed to ResultSystem with the current Enemy object.
Player::shoot() / skip()
04
Result Evaluated
ResultSystem::evaluate() checks isThreat() against the action. Score is updated (+10 correct shoot, +5 correct skip) or a life is lost.
ResultSystem::evaluate()
05
Outcome Shown
Win/Lose feedback is displayed. If lives reach 0 or all 5 waves clear, the game ends and a rank (S/A/B/C) is awarded.
ResultSystem::display()
06
Play Again
Player can restart — all objects are re-initialised from scratch, demonstrating proper object lifecycle management.
Re-init all objects

DOCUMENTATION

Project Overview
What is One Bullet?
  • ▸ A decision-making game where the player has exactly one bullet per round to use — or not.
  • ▸ Enemies appear one at a time. The player must identify if the enemy is a threat and react correctly.
  • ▸ Built in C++ for console, with a full HTML5 interactive demo for presentation.
  • ▸ Designed to demonstrate OOP concepts from the OODP course in a fun, tangible format.
Design Decisions
Architecture Notes
  • ▸ Struct vs Class: Enemy uses a struct for lightweight data grouping; Player and ResultSystem use classes with access control.
  • ▸ Inheritance: DangerousEnemy and HarmlessEnemy extend Enemy to illustrate polymorphic dispatch cleanly.
  • ▸ ResultSystem: Separated from Game intentionally — single responsibility principle keeps evaluation logic isolated.
  • ▸ Randomisation: srand(time(0)) ensures a fresh enemy order each game, adding replayability.
OOP Mapping
Concepts Covered
  • ▸ Classes & Objects — Game, Player, Enemy, ResultSystem
  • ▸ Encapsulation — private members with public accessors in Player
  • ▸ Inheritance — DangerousEnemy, HarmlessEnemy extend Enemy
  • ▸ Polymorphism — virtual isThreat() overridden by subtypes
